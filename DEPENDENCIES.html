Configuration Notes:

You must configure the documentation runner in the file navadoc.xml in the [project home]/NavaDoc/config directory.  Note there is a section to configure the runner, plus the Log4J section to configure the amount, targets and formatting of the facility's logging output.
The following is an example:

<navadoc>
  <configuration>

      <property name="services-path" 
		description="path to Navajo Services XML files" 
		value="/home/meichler/projects/sportlink-serv/navajo-tester/auxilary/scripts" />

      <property name="stylesheet-path"
		description="path to XLST style sheet for Navajo services XML formatting"
		value="/home/meichler/projects/NavaDoc/xsl/navadoc.xsl" />

      <property name="target-path"
		description="target directory for documentation"
		value="/home/meichler/projects/sportlink-serv/doc" />

  </configuration>

  <log4j:configuration xmlns:log4j="http://jakarta.apache.org/log4j/">

      <appender name="R" class="org.apache.log4j.RollingFileAppender">
	  <param name="File"           value="/home/meichler/projects/NavaDoc/log/navadoc.log" />
	  <param name="MaxFileSize"    value="100KB"/>
	  <param name="MaxBackupIndex" value="1"/>
	  <layout class="org.apache.log4j.PatternLayout">
	      <param name="ConversionPattern" value="%d %-5p [%t] %C{2} (%F:%L) - %m%n"/>
	  </layout>       
      </appender>

      <appender name="STDOUT" class="org.apache.log4j.ConsoleAppender">
	      <layout class="org.apache.log4j.PatternLayout">
		 <param name="ConversionPattern"
			value="%d %-5p [%t] %C{2} (%F:%L) - %m%n"/>
	      </layout>               
      </appender>

      <category name="org.apache.log4j.xml">
	<priority value="debug" />
	<appender-ref ref="R" />
      </category>

      <root>
	 <priority value ="debug" />
	 <appender-ref ref="STDOUT" />
      </root>

  </log4j:configuration>
</navadoc>
